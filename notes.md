    <ul>

            <li>Emotional Stability</li>

            <li>Rule-Consciousness</li>
            <li>Social Boldness</li>
            <li>Sensitivity</li>
            <li>Vigilance</li>
            <li>Abstractedness</li>
            <li>Privateness</li>
            <li>Apprehension</li>
            <li>Openness to Change</li>
            <li>Self-Reliance</li>
            <li>Perfectionism</li>
            <li>Tension</li>
        </ul>
